action int fibonacci(int n) {
    if (n <= 0) {
        print "Please enter a positive integer";
        addline;
        report 0;
    }
    elif (n == 1) {
        report 0;
    }
    elif (n == 2) {
        report 1;
    }
    else {
        declare left as int;
        declare right as int;
        let left = n - 1;
        let right = n - 2;
        report fibonacci(left) + fibonacci(right);
    }
    # Default report if all else fails
    report 0;
}

declare test as int;
let test = fibonacci(3);
print "Fibonacci Result: ";
print test;
addline;
print "Complete!";
